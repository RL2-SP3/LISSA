 41def FailureMerge(baseData: pd.DataFrame ,dirFile:str ,fileName:str) -> Tuple[pd.DataFrame,list]:
 42
 43    '''
 44    Then, the file respostas.csv is called. respostas.csv is an PEREGRINO ESP Run Life fail spreadsheet.
 45
 46    This file is loaded with the intent of allowing our data be linked with faliure date.
 47
 48    A column, called "Well_down" was created by Equinor, analyzing if the well was on off or online. 
 49
 50    Having the faliure date at main data, it is possible to realize a comparision between the dates and infer if there will be a faliure or not. Therefore, with this approach, we can label all data, if it's relatable with a faliure or not.
 51
 52    However, this approach does not point exactly when it was observed the faliure, but, in reality, only labels if the proababilty of the anomaly is inside the subset is greater.
 53
 54    '''
 55
 56    filePath = dirFile + "/" + fileName #path to file
 57    respostas = pd.read_csv(filePath, delimiter=",") #import faliure data
 58
 59    pumpList = baseData["Well Run"].unique() #filter all pump names at the dataset
 60
 61    
 62    respostas = respostas.loc[respostas["Well Run"].isin(pumpList)] #only analyzes data that is at the time dataset
 63
 64    #faliure precise time is not considered on the dataset, since its only a report.
 65    respostas["Failure Date"] = pd.to_datetime(respostas["Failure Date"],format="%d/%m/%y").dt.tz_localize('UTC')
 66
 67    entireData = pd.merge(baseData,respostas,how="left",on="Well Run") #do a left merge to unite failiure date with time info
 68
 69    entireData["Failure"] = entireData["time"] >= entireData["Failure Date"] #check if failiure already happened
 70
 71    dropList = ['Start-up Date','Start-up Date','ESP Failure?','Failure Date'] 
 72
 73    entireData.drop(dropList,axis=1,inplace=True) #remove duplicated columns
 74
 75    # pumpData.drop(entireData.loc[baseData["Well_down"]==True].index,inplace=True) #offline data is dropped
 76    # entireData.drop(columns="Well_down",inplace=True) #we dont need anymore to know when the well is down
 77
 78    entireData.set_index("time",inplace=True) #time is set as index
 79
 80    return entireData, pumpList

255def ProcessData(pumpList: list ,entireData: pd.DataFrame ,windowSize=1,totalDataPath="/")->pd.DataFrame:
256    '''
257
258    Apply filter to all pumps and export the resulting dataset.
259    
260    '''
261    totalData = pd.DataFrame(columns=list(entireData))

317'''
318General things here:
319
320All filters are normalized here, this means that filter return is f(x) = (H(x) - mean(x))/std(x). 
321A separated approach was tried, but the results were strange.
322As a standard, all windows sizes are set in matter of days. If Equinor releases full granularity data, we will be allowed to change this parameter.
323'''
324
325#High Frequency Filter -> H_high= 1 - H_low
326def HighFrequencyFilter(x,windowSize):
327    return (x-x.ewm(span=24*windowSize).mean()-x.expanding().mean())/x.expanding().std()
328
329#Low Frequency Filter
330def LowFrequencyFilter(x,windowSize):
331    return (x.ewm(span=24*windowSize).mean()-x.expanding().mean())/x.expanding().std()
332
333#Mean Average Filter
334def MeanAverageFilter(x,windowSize):
335    return (x.rolling(24*windowSize,min_periods=0).mean()-x.expanding().mean())/x.expanding().std()
336
337#Standard filter
338def StandardizerFilter(x):
339    return (x-x.expanding().mean())/x.expanding().std()

346def FilterCreation(exportData,windowSize,filterType):
347
348    groupData = exportData.copy()
349    groupData["group"] = (exportData["Well_down"] != exportData["Well_down"].shift()).cumsum()
350    groupData = groupData.groupby('group')
351
352    Filter = groupData[relevantHeader(exportData)].apply(lambda x: filterType(x,windowSize))
353    
354    return (Filter
355        .reset_index()
356        .set_index("time")
357        .drop(columns="group")
358        .fillna(0))
